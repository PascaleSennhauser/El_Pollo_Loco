class DrawableObject {
    x = 120;
    y = 330;
    height = 100;
    width = 100;
    offset = {
        top: 0,
        left: 0,
        right: 0,
        bottom: 0
    };
    img;
    imageCache = {};
    currentImage = 0;



    loadImage(path) {
        this.img = new Image();
        this.img.src = path;
    }


    draw(ctx) {
        ctx.drawImage(this.img, this.x, this.y, this.width, this.height);
    }


    /**
     * 
     * @param {Array} arr - ['img/image1.png': img, 'img/image2.png': img, ...]
     */
        loadImages(arr) {
            arr.forEach((path) => {
                let img = new Image();
                img.src = path;
                this.imageCache[path] = img;
            })
        }
